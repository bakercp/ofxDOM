620 template <typename ElementType>
621 ElementType* Element::addChild(std::unique_ptr<ElementType> element)
622 {
623  static_assert(std::is_base_of<Element, ElementType>(), "ElementType must be an Element or derived from Element.");
624 
625  if (element)
626  {
627  // Get a raw pointer to the node for later.
628  ElementType* pNode = element.get();
629 
630  // Assign the parent to the node via the raw pointer.
631  pNode->_parent = this;
632 
633  // Take ownership of the node.
634  _children.push_back(std::move(element));
635 
636  // Invalidate all cached child geometry.
637  invalidateChildGeometry();
638 
639  // Alert the node that its parent was set.
640  ElementEventArgs addedEvent(this);
641  ofNotifyEvent(pNode->addedTo, addedEvent, this);
642 
643  ElementEventArgs childAddedEvent(pNode);
644  ofNotifyEvent(childAdded, childAddedEvent, this);
645 
646  // Attach child listeners.
647  ofAddListener(pNode->move, this, &Element::_onChildMoved);
648  ofAddListener(pNode->resize, this, &Element::_onChildResized);
649 
651  for (auto& child : _children)
652  {
653  // Don't alert itself.
654  if (child.get() != pNode)
655  {
656  ElementEventArgs event(pNode);
657  ofNotifyEvent(child->siblingAdded, event, this);
658  }
659  }
660 
661  return pNode;
662  }
663  else
664  {
665  return nullptr;
666  }
667 }

670 template <typename ElementType>
671 std::vector<ElementType*> Element::siblings()
672 {
673  static_assert(std::is_base_of<Element, ElementType>(), "ElementType must be an Element or derived from Element.");
674 
675  std::vector<ElementType*> results;
676 
677  if (_parent)
678  {
679  for (auto& child : _parent->_children)
680  {
681  ElementType* pChild = dynamic_cast<ElementType*>(child.get());
682 
683  if (pChild != this)

685  results.push_back(pChild);
686  }
687  }
688  }

748 template <typename AnyType>
749 AnyType Element::getAttribute(const std::string& key, bool inherit) const
750 {
751  auto iter = _attributes.find(key);
752 
753  if (iter != _attributes.end() && iter->second.is<AnyType>())
754  {
755  return iter->second.as<AnyType>();
756  }
757  else if (inherit && hasParent())
758  {
759  return parent()->getAttribute<AnyType>(key);
760  }
761  else
762  {
763  throw DOMException(DOMException::INVALID_ATTRIBUTE_KEY);
764  }
