364 template <typename EventArgsType>
365 class DOMEvent: public BaseDOMEvent
366 {
367 public:
368  virtual ~DOMEvent()
369  {
370  }
371 
372  bool hasBubblePhaseListeners() const override
373  {
374  return _bubbleEvent.size() > 0;
375  }
376 
377  bool hasCapturePhaseListeners() const override
378  {
379  return _captureEvent.size() > 0;
380  }
381 
382  ofEvent<EventArgsType>& event(bool useCapture = false)
383  {
384  return useCapture ? _captureEvent : _bubbleEvent;
385  }
386 
387  void notify(EventArgsType& e)
388  {
389  switch (e.getPhase())
390  {
391  case EventArgs::Phase::NONE:
392  throw DOMException(DOMException::INVALID_STATE_ERROR + ": " + "DOMEvent::notify");
393  case EventArgs::Phase::CAPTURING_PHASE:
394  _captureEvent.notify(e.source(), e);
395  return;
396  case EventArgs::Phase::AT_TARGET:
397  _captureEvent.notify(e.source(), e);
398  _bubbleEvent.notify(e.source(), e);
399  return;
400  case EventArgs::Phase::BUBBLING_PHASE:
401  _bubbleEvent.notify(e.source(), e);
402  return;
403  }
404  }
405 
406 private:
407  ofEvent<EventArgsType> _bubbleEvent;
408  ofEvent<EventArgsType> _captureEvent;
409
